<input type="checkbox" name="gfc" id="good" value="good" onclick="check1()"> <label for="good">good</label><br>
<input type="checkbox" name="gfc" id="fast" value="fast" onclick="check2()"> <label for="fast">fast</label><br>

<script>
    function check1() {
        let good = document.getElementById("good").checked;
        let fast = document.getElementById("fast").checked;
        if (good === fast) {
            document.getElementById("cheap").checked = false;
        }
    }
    function check2() {
        let fast = document.getElementById("fast").checked;
        let cheap = document.getElementById("good").checked;
        if (fast === cheap) {
            document.getElementById("good").checked = false;
        }
    }
    function check3() {
        let cheap = document.getElementById("cheap").checked;
        let fast = document.getElementById("fast").checked;
        if (cheap === fast) {
            document.getElementById("good").checked = false;
        }
    }
</script>
